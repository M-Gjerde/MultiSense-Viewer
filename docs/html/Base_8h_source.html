93 void updateUniformBufferData(Base::Render d, ScriptType scriptType) {
94 this->renderData = d;
95
96 update();
97 // If initialized
98 if (renderUtils.uniformBuffers.empty())
99 return;
100
101 UniformBufferSet currentUB = renderUtils.uniformBuffers[renderData.index];
102 if (scriptType == ArDefault || scriptType == AR_POINT_CLOUD) {
103 // TODO unceesarry mapping and unmapping occurring here.
104 currentUB.bufferOne.map();
105 memcpy(currentUB.bufferOne.mapped, bufferOneData, sizeof(UBOMatrix));
106 currentUB.bufferOne.unmap();
107
108 currentUB.bufferTwo.map();
109 memcpy(currentUB.bufferTwo.mapped, bufferTwoData, sizeof(FragShaderParams));
110 currentUB.bufferTwo.unmap();
111 if (bufferThreeData == NULL)
112 return;
113
114 currentUB.bufferThree.map();
115 char *val = static_cast<char *>(bufferThreeData);
116 float f = (float) atoi(val);
117 memcpy(currentUB.bufferThree.mapped, &f, sizeof(float));
118 currentUB.bufferThree.unmap();
119 } else if(scriptType == AR_POINT_CLOUD){
120 currentUB.bufferOne.map();
121 memcpy(currentUB.bufferOne.mapped, bufferOneData, sizeof(UBOMatrix));
122 currentUB.bufferOne.unmap();
123
124 currentUB.bufferTwo.map();
125 memcpy(currentUB.bufferTwo.mapped, bufferTwoData, sizeof(PointCloudShader));
126 currentUB.bufferTwo.unmap();
127 }

132 void createUniformBuffers(RenderUtils utils, ScriptType scriptType) {
133 if (scriptType == ArDisabled)
134 return;
135
136 renderUtils = std::move(utils);
137 renderUtils.uniformBuffers.resize(renderUtils.UBCount);
138
139 if (scriptType == ArDefault || scriptType == AR_POINT_CLOUD) {
140
141 bufferOneData = new UBOMatrix();
142 bufferTwoData = new FragShaderParams();
143 bufferThreeData = new float();
144
145 for (auto &uniformBuffer: renderUtils.uniformBuffers) {
146
147 renderUtils.device->createBuffer(VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
148 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
149 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
150 &uniformBuffer.bufferOne, sizeof(UBOMatrix));
151
152 renderUtils.device->createBuffer(VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
153 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
154 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
155 &uniformBuffer.bufferTwo, sizeof(FragShaderParams));
156
157 renderUtils.device->createBuffer(VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
158 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
159 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
160 &uniformBuffer.bufferThree, sizeof(float));
161
162 }
163 }
164 else if (scriptType == AR_POINT_CLOUD) {
165 bufferOneData = new UBOMatrix();
166 bufferTwoData = new PointCloudShader();
167
168 for (auto &uniformBuffer: renderUtils.uniformBuffers) {
169
170 renderUtils.device->createBuffer(VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT,
171 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
172 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
173 &uniformBuffer.bufferOne, sizeof(UBOMatrix));

176 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |

178 &uniformBuffer.bufferTwo, sizeof(PointCloudShader));
179 }
180 }

185 [[nodiscard]] VkPipelineShaderStageCreateInfo
186 loadShader(std::string fileName, VkShaderStageFlagBits stage) const {
187
188 // Check if we have .spv extensions. If not then add it.
189 std::size_t extension = fileName.find(".spv");
190 if (extension == std::string::npos)
191 fileName.append(".spv");
192
193
194 VkPipelineShaderStageCreateInfo shaderStage = {};
195 shaderStage.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
196 shaderStage.stage = stage;
197 shaderStage.module = Utils::loadShader((Utils::getShadersPath() + fileName).c_str(),
198 renderUtils.device->logicalDevice);
199 shaderStage.pName = "main";
200 assert(shaderStage.module != VK_NULL_HANDLE);
201 // TODO CLEANUP SHADERMODULES WHEN UNUSED
202 return shaderStage;
203 }
